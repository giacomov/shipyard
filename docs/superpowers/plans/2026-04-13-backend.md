# Shipyard Backend Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the FastAPI backend for Shipyard — a Kanban board API with PostgreSQL persistence, JWT mock authentication, card dependency enforcement, and real-time WebSocket broadcasting.

**Architecture:** FastAPI app with async SQLAlchemy ORM and Alembic migrations. Business logic lives in a `services/` layer; routers are thin. A singleton WebSocket `ConnectionManager` holds open connections per project and is called by REST handlers after successful mutations. All dependency rules are enforced server-side in `services/dependency_service.py`.

**Tech Stack:** Python 3.13, FastAPI, SQLAlchemy 2.0 (async + asyncpg), Alembic, Pydantic v2, python-jose, bcrypt, pytest, pytest-asyncio, httpx

```python
import uuid
from datetime import datetime

from sqlalchemy import Column as SAColumn
from sqlalchemy import DateTime, ForeignKey, Integer, String, Table, Text, func
from sqlalchemy.orm import Mapped, mapped_column, relationship

from .base import Base

# Association table for card dependencies
card_dependencies = Table(
    "card_dependencies",
    Base.metadata,
    SAColumn("card_id", ForeignKey("cards.id", ondelete="CASCADE"), primary_key=True),
    SAColumn(
        "depends_on_id", ForeignKey("cards.id", ondelete="CASCADE"), primary_key=True
    ),
)


class Card(Base):
    __tablename__ = "cards"

    id: Mapped[uuid.UUID] = mapped_column(primary_key=True, default=uuid.uuid4)
    column_id: Mapped[uuid.UUID] = mapped_column(ForeignKey("columns.id"))
    project_id: Mapped[uuid.UUID] = mapped_column(ForeignKey("projects.id"))
    title: Mapped[str] = mapped_column(String(500))
    description: Mapped[str | None] = mapped_column(Text, nullable=True)
    position: Mapped[int] = mapped_column(Integer)
    created_by: Mapped[uuid.UUID] = mapped_column(ForeignKey("users.id"))
    created_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True), server_default=func.now()
    )

    column: Mapped["Column"] = relationship(back_populates="cards")
    creator: Mapped["User"] = relationship(back_populates="created_cards")

    # Cards this card depends on (must be Done before this card can leave backlog)
    depends_on: Mapped[list["Card"]] = relationship(
        secondary=card_dependencies,
        primaryjoin=id == card_dependencies.c.card_id,
        secondaryjoin=id == card_dependencies.c.depends_on_id,

    # Cards that depend on this card
    dependents: Mapped[list["Card"]] = relationship(
        secondary=card_dependencies,

        secondaryjoin=id == card_dependencies.c.card_id,
```

```python
import asyncio
from logging.config import fileConfig

from alembic import context
from sqlalchemy.ext.asyncio import create_async_engine

from app.core.config import settings
from app.models import Base  # noqa: F401 — imports all models so metadata is populated

config = context.config
if config.config_file_name is not None:
    fileConfig(config.config_file_name)

target_metadata = Base.metadata


def run_migrations_offline() -> None:
    context.configure(
        url=settings.database_url,
        target_metadata=target_metadata,
        literal_binds=True,
        dialect_opts={"paramstyle": "named"},
    )
    with context.begin_transaction():
        context.run_migrations()


async def run_migrations_online() -> None:
    connectable = create_async_engine(settings.database_url)
    async with connectable.connect() as connection:
        await connection.run_sync(
            lambda conn: context.configure(
                connection=conn, target_metadata=target_metadata
            )
        )
        async with connection.begin():
            await connection.run_sync(lambda _: context.run_migrations())
    await connectable.dispose()


if context.is_offline_mode():
    run_migrations_offline()
else:
    asyncio.run(run_migrations_online())
```

```python
from datetime import datetime, timedelta, timezone

from jose import jwt

from .config import settings


def create_access_token(user_id: str) -> str:
    expire = datetime.now(timezone.utc) + timedelta(
        minutes=settings.access_token_expire_minutes
    )
    return jwt.encode(
        {"sub": user_id, "exp": expire, "type": "access"},
        settings.jwt_secret,
        algorithm=settings.jwt_algorithm,
    )


def create_refresh_token(user_id: str) -> str:
    expire = datetime.now(timezone.utc) + timedelta(
        days=settings.refresh_token_expire_days
    )
    return jwt.encode(
        {"sub": user_id, "exp": expire, "type": "refresh"},
        settings.jwt_secret,
        algorithm=settings.jwt_algorithm,
    )


def decode_token(token: str) -> dict:
    return jwt.decode(
        token, settings.jwt_secret, algorithms=[settings.jwt_algorithm]
    )
```

- [ ] **Step 2: Create `backend/app/core/mock_seed.py`**

Seeds users from `MOCK_USERS` env var into the database at startup (idempotent).

```python
import bcrypt
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from ..models.user import User
from .config import settings
from .database import AsyncSessionLocal


async def seed_mock_users() -> None:
    async with AsyncSessionLocal() as db:
        for entry in settings.mock_users_list:
            result = await db.execute(
                select(User).where(User.email == entry["email"])
            )
            if result.scalar_one_or_none() is not None:
                continue
            hashed = bcrypt.hashpw(
                entry["password"].encode(), bcrypt.gensalt()
            ).decode()
            db.add(
                User(
                    email=entry["email"],
                    display_name=entry["display_name"],
                    hashed_password=hashed,
                )
            )
        await db.commit()
```

```python
import uuid

from fastapi import Depends, HTTPException, status
from fastapi.security import HTTPAuthorizationCredentials, HTTPBearer
from jose import JWTError
from sqlalchemy.ext.asyncio import AsyncSession

from ..core.auth import decode_token
from ..core.database import get_db
from ..models.user import User
from ..services.auth_service import get_user_by_id

bearer = HTTPBearer()


async def get_current_user(
    credentials: HTTPAuthorizationCredentials = Depends(bearer),
    db: AsyncSession = Depends(get_db),
) -> User:
    exc = HTTPException(
        status_code=status.HTTP_401_UNAUTHORIZED,
        detail="Invalid or expired token",
    )
    try:
        payload = decode_token(credentials.credentials)
        if payload.get("type") != "access":
            raise exc
        raw_id = payload.get("sub")
        if not raw_id:
            raise exc
    except JWTError:
        raise exc

    user = await get_user_by_id(db, uuid.UUID(raw_id))
    if not user:
        raise exc
    return user
```

- [ ] **Step 7: Create `backend/app/api/auth.py`**

```python
from fastapi import APIRouter, Depends, HTTPException, status
from jose import JWTError
from sqlalchemy.ext.asyncio import AsyncSession

from ..core.auth import create_access_token, create_refresh_token, decode_token
from ..core.database import get_db
from ..schemas.auth import (
    AccessTokenResponse,
    LoginRequest,
    RefreshRequest,
    TokenResponse,
)
from ..schemas.user import UserOut
from ..services.auth_service import get_user_by_email, get_user_by_id, verify_password

router = APIRouter()


@router.post("/login", response_model=TokenResponse)
async def login(body: LoginRequest, db: AsyncSession = Depends(get_db)) -> TokenResponse:
    user = await get_user_by_email(db, body.email)
    if not user or not verify_password(body.password, user.hashed_password):
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Invalid credentials",
        )
    return TokenResponse(
        access_token=create_access_token(str(user.id)),
        refresh_token=create_refresh_token(str(user.id)),
    )


@router.post("/refresh", response_model=AccessTokenResponse)
async def refresh(body: RefreshRequest, db: AsyncSession = Depends(get_db)) -> AccessTokenResponse:
    exc = HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Invalid refresh token")
    try:
        payload = decode_token(body.refresh_token)
        if payload.get("type") != "refresh":
            raise exc
        import uuid
        user = await get_user_by_id(db, uuid.UUID(payload["sub"]))
        if not user:
            raise exc
    except (JWTError, Exception):
        raise exc
    return AccessTokenResponse(access_token=create_access_token(str(user.id)))
```

```python
from ..models.user import User

async def create_project(
    db: AsyncSession, creator: User, data: ProjectCreate
) -> tuple[Project, ProjectMember]:
    project = Project(
        name=data.name,
        description=data.description,
        created_by=creator.id,
    )
    db.add(project)
    await db.flush()  # get project.id before commit

    membership = ProjectMember(
        project_id=project.id, user_id=creator.id, role="owner"
    )
    db.add(membership)
    await db.commit()
    await db.refresh(project)
    return project, membership


async def list_user_projects(
    db: AsyncSession, user_id: uuid.UUID
) -> list[tuple[Project, str]]:
    result = await db.execute(
        select(Project, ProjectMember.role)
        .join(ProjectMember, ProjectMember.project_id == Project.id)
        .where(ProjectMember.user_id == user_id)
        .order_by(Project.created_at.desc())
    )
    return result.all()


async def get_project_or_403(
    db: AsyncSession, project_id: uuid.UUID, user_id: uuid.UUID
) -> tuple[Project, str]:
    """Returns (project, role) or raises 403."""
    result = await db.execute(
        select(Project, ProjectMember.role)
        .join(ProjectMember, ProjectMember.project_id == Project.id)
        .where(Project.id == project_id, ProjectMember.user_id == user_id)
        .options(
            selectinload(Project.members).selectinload(ProjectMember.user),
            selectinload(Project.columns),
        )
    )
    row = result.first()
    if row is None:
        raise HTTPException(status_code=status.HTTP_403_FORBIDDEN, detail="Not a project member")
    return row


async def add_member(
    db: AsyncSession, project_id: uuid.UUID, email: str, requester_role: str
) -> ProjectMember:
    if requester_role != "owner":
        raise HTTPException(status_code=403, detail="Only owners can add members")
    result = await db.execute(select(User).where(User.email == email))
    user = result.scalar_one_or_none()
    if not user:
        raise HTTPException(status_code=404, detail="User not found")
    membership = ProjectMember(project_id=project_id, user_id=user.id, role="member")
    db.add(membership)
    await db.commit()
    await db.refresh(membership)
    return membership


async def remove_member(
    db: AsyncSession, project_id: uuid.UUID, target_user_id: uuid.UUID, requester_role: str
) -> None:

        select(ProjectMember).where(
            ProjectMember.project_id == project_id,
            ProjectMember.user_id == target_user_id,
        )
    )
    membership = result.scalar_one_or_none()
    if not membership:
        raise HTTPException(status_code=404, detail="Member not found")
```

```python
import uuid

from fastapi import HTTPException
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from ..models.column import Column
from ..schemas.column import ColumnCreate, ColumnReorderItem, ColumnUpdate


async def add_column(
    db: AsyncSession, project_id: uuid.UUID, data: ColumnCreate
) -> Column:
    result = await db.execute(
        select(Column)
        .where(Column.project_id == project_id)
        .order_by(Column.position.desc())
        .limit(1)
    )
    last = result.scalar_one_or_none()
    next_pos = (last.position + 1) if last else 0
    col = Column(project_id=project_id, name=data.name, position=next_pos)
    db.add(col)
    await db.commit()
    await db.refresh(col)
    return col


async def rename_column(
    db: AsyncSession, column_id: uuid.UUID, data: ColumnUpdate, user_project_ids: set[uuid.UUID]
) -> Column:
    result = await db.execute(select(Column).where(Column.id == column_id))
    col = result.scalar_one_or_none()
    if not col or col.project_id not in user_project_ids:
        raise HTTPException(status_code=404, detail="Column not found")
    col.name = data.name
    await db.commit()
    await db.refresh(col)
    return col


async def reorder_columns(
    db: AsyncSession, project_id: uuid.UUID, items: list[ColumnReorderItem]
) -> list[Column]:
    result = await db.execute(select(Column).where(Column.project_id == project_id))
    cols_by_id = {c.id: c for c in result.scalars().all()}
    for item in items:
        if item.id in cols_by_id:
            cols_by_id[item.id].position = item.position
    await db.commit()
    return sorted(cols_by_id.values(), key=lambda c: c.position)


async def delete_column(
    db: AsyncSession, column_id: uuid.UUID, user_project_ids: set[uuid.UUID]
) -> None:
    result = await db.execute(select(Column).where(Column.id == column_id))
    col = result.scalar_one_or_none()
```

```python
import uuid

from fastapi import HTTPException
from sqlalchemy import delete, select
from sqlalchemy.ext.asyncio import AsyncSession

from ..models.card import Card, card_dependencies
from ..models.column import Column


async def would_create_cycle(
    db: AsyncSession, card_id: uuid.UUID, new_dep_id: uuid.UUID
) -> bool:
    """
    Returns True if adding (card_id depends_on new_dep_id) would create a cycle.
    Uses BFS from new_dep_id through the depends_on graph; cycle exists if we reach card_id.
    """
    visited: set[uuid.UUID] = set()
    frontier = [new_dep_id]
    while frontier:
        current = frontier.pop()
        if current == card_id:
            return True
        if current in visited:
            continue
        visited.add(current)
        result = await db.execute(
            select(card_dependencies.c.depends_on_id).where(
                card_dependencies.c.card_id == current
            )
        )
        frontier.extend(result.scalars().all())
    return False


async def add_dependency(
    db: AsyncSession, card_id: uuid.UUID, dep_id: uuid.UUID
) -> None:
    if card_id == dep_id:
        raise HTTPException(status_code=422, detail="A card cannot depend on itself")
    if await would_create_cycle(db, card_id, dep_id):
        raise HTTPException(status_code=422, detail="This would create a circular dependency")
    await db.execute(
        card_dependencies.insert().values(card_id=card_id, depends_on_id=dep_id)
    )
    await db.commit()


async def remove_dependency(
    db: AsyncSession, card_id: uuid.UUID, dep_id: uuid.UUID
) -> None:
    result = await db.execute(
        delete(card_dependencies).where(
            card_dependencies.c.card_id == card_id,
            card_dependencies.c.depends_on_id == dep_id,
        )
    )
    if result.rowcount == 0:
        raise HTTPException(status_code=404, detail="Dependency not found")
    await db.commit()


async def validate_move(
    db: AsyncSession,
    card: Card,
    target_column_id: uuid.UUID,
    target_position: int,
) -> None:
    """
    Enforces Rule 1 and Rule 2. Raises HTTP 422 on violation.

    Rule 1: A card can only leave the leftmost column (backlog) if ALL its
            dependencies are in the rightmost column (Done).

    Rule 2: Within the same column, if moving UP (lower position), the card
            cannot end up above any dependency that is currently above it
            (i.e. any dep in [target_position, card.position - 1]).
    """
    # Load all columns for this project to find backlog and done
    result = await db.execute(
        select(Column)
        .where(Column.project_id == card.project_id)
        .order_by(Column.position)
    )
    cols = result.scalars().all()
    if not cols:
        return

    backlog_id = cols[0].id
    done_id = cols[-1].id

    # Rule 1
    if card.column_id == backlog_id and target_column_id != backlog_id:
        for dep in card.depends_on:
            if dep.column_id != done_id:
                raise HTTPException(
                    status_code=422,
                    detail=f"'{card.title}' depends on '{dep.title}' which is not in Done",
                )

    # Rule 2: only applies when reordering within the same column moving upward
    if target_column_id == card.column_id and target_position < card.position:
        for dep in card.depends_on:
            if dep.column_id == card.column_id and dep.position < card.position:
                if dep.position >= target_position:
                    raise HTTPException(
                        status_code=422,
                        detail=(
                            f"'{card.title}' depends on '{dep.title}' "
                            f"— '{dep.title}' must come first"
                        ),
                    )
```

```python
from sqlalchemy import update as sa_update
from .dependency_service import validate_move


async def move_card(
    db: AsyncSession,
    card_id: uuid.UUID,
    target_column_id: uuid.UUID,
    target_position: int,
) -> Card:
    card = await _get_card_or_404(db, card_id)
    await validate_move(db, card, target_column_id, target_position)

    old_column_id = card.column_id
    old_position = card.position

    if old_column_id == target_column_id:
        # Reorder within same column
        if target_position < old_position:
            # Moving up: shift cards in [target_position, old_position - 1] down
            await db.execute(
                sa_update(Card)
                .where(
                    Card.column_id == old_column_id,
                    Card.position >= target_position,
                    Card.position < old_position,
                    Card.id != card_id,
                )
                .values(position=Card.position + 1)
            )
        elif target_position > old_position:
            # Moving down: shift cards in [old_position + 1, target_position] up
            await db.execute(
                sa_update(Card)
                .where(
                    Card.column_id == old_column_id,
                    Card.position > old_position,
                    Card.position <= target_position,
                    Card.id != card_id,
                )
                .values(position=Card.position - 1)
            )
    else:
        # Moving between columns: close the gap in the old column
        await db.execute(
            sa_update(Card)
            .where(Card.column_id == old_column_id, Card.position > old_position)
            .values(position=Card.position - 1)
        )
        # Open a gap in the target column
        await db.execute(
            sa_update(Card)
            .where(Card.column_id == target_column_id, Card.position >= target_position)
            .values(position=Card.position + 1)
        )
        card.column_id = target_column_id

    card.position = target_position
    await db.commit()
    await db.refresh(card)
    return card
```

- [ ] **Step 1: Create `backend/app/ws/manager.py`**

```python
import json
from collections import defaultdict

from fastapi import WebSocket


class ConnectionManager:
    def __init__(self) -> None:
        self._connections: dict[str, set[WebSocket]] = defaultdict(set)

    def add(self, ws: WebSocket, project_id: str) -> None:
        self._connections[project_id].add(ws)

    def remove(self, ws: WebSocket, project_id: str) -> None:
        self._connections[project_id].discard(ws)

    async def broadcast(self, project_id: str, event: dict) -> None:
        dead: set[WebSocket] = set()
        for ws in list(self._connections[project_id]):
            try:
                await ws.send_json(event)
            except Exception:
                dead.add(ws)
        for ws in dead:
            self.remove(ws, project_id)


manager = ConnectionManager()
```

- [ ] **Step 2: Create `backend/app/ws/router.py`**

```python
import asyncio

from fastapi import APIRouter, WebSocket, WebSocketDisconnect
from jose import JWTError

from ..core.auth import decode_token
from .manager import manager

router = APIRouter()


@router.websocket("/ws/projects/{project_id}")
async def board_ws(websocket: WebSocket, project_id: str) -> None:
    await websocket.accept()

    # First-message auth handshake (10-second timeout)
    try:
        raw = await asyncio.wait_for(websocket.receive_json(), timeout=10.0)
        if raw.get("type") != "auth":
            await websocket.close(code=4001, reason="Expected auth message")
            return
        payload = decode_token(raw.get("token", ""))
        if payload.get("type") != "access":
            await websocket.close(code=4001, reason="Invalid token type")
            return
    except (JWTError, asyncio.TimeoutError, Exception):
        await websocket.close(code=4001, reason="Auth failed")
        return

    await websocket.send_json({"type": "auth.ok"})
    manager.add(websocket, project_id)

    try:
        while True:
            # Keep the connection alive; client sends no messages after auth
            await websocket.receive_text()
    except WebSocketDisconnect:
        manager.remove(websocket, project_id)
```
